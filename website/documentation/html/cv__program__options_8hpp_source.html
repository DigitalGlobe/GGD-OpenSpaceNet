198 template<class channelT, class charT>
199 void validate(boost::any& v, const std::vector <std::basic_string<charT>>& values, Rect_<channelT>*, int)
200 {
201  // Make sure no previous assignment to 'v' was made.
202  boost::program_options::validators::check_first_occurrence(v);
203 
204  // Combine the values, then split again so that we have a full list of arguments in the end
205  auto combined = boost::join(values, " ");
206 
207  std::vector <std::basic_string<charT>> splitValues;
208  boost::split(splitValues, combined, boost::is_any_of(",;:/ "), boost::token_compress_on);
209 
210  if(splitValues.size() != 4) {
211  throw boost::program_options::validation_error(boost::program_options::validation_error::invalid_option_value);
212  }
213 
214  std::vector <channelT> nums;
215  for(const auto& value : splitValues) {
216  nums.push_back(boost::lexical_cast<channelT>(value));
217  }
218 
219  v = boost::any(cv::Rect_<channelT>(cv::Point_<channelT>(nums[0], nums[1]), cv::Point_<channelT>(nums[2], nums[3])));
220 }

227 template<class coordT, class charT>
228 void validate(boost::any& v, const std::vector <std::basic_string<charT>>& values, coordT* target_type, int)
229 {
230  typedef typename cv::DataType<coordT>::channel_type channel_type;
231 
232  // Make sure no previous assignment to 'v' was made.
233  boost::program_options::validators::check_first_occurrence(v);
234 
235  // Combine the values, then split again so that we have a full list of arguments in the end
236  auto combined = boost::join(values, " ");
237 
238  std::vector <std::basic_string<charT>> splitValues;
239  boost::split(splitValues, combined, boost::is_any_of(",;:/ "), boost::token_compress_on);
240 
241  if(!splitValues.size()) {
242  throw boost::program_options::validation_error(boost::program_options::validation_error::at_least_one_value_required);
243  }
244 
245  std::vector <channel_type> nums;
246  for(const auto& value : splitValues) {
247  nums.push_back(boost::lexical_cast<channel_type>(value));
248  }
249 
250  if(nums.size() < 2) {
251  v = boost::any(coordT(nums[0], nums[0]));
252  } else {
253  v = boost::any(coordT(nums[0], nums[1]));
254  }
255 }
