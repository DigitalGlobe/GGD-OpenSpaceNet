171 template <typename T>
172 void PyramidImageBatch<T>::init(const cv::Mat& image, const PyramidSlidingWindow& pyramidSlidingWindow, size_t batchSize, T* outData, size_t outDataSize, const cv::Size& windowSize)
173 {
174  batchSize_ = batchSize;
175  data_ = outData;
176  dataSize_ = outDataSize;
177 
178  if(pyramidSlidingWindow.slidingWindows()[0].edgeBehavior() == EdgeBehavior::CROP) {
179  DG_ERROR_THROW("Sliding windows with CROP edge behavior are not supported");
180  }
181 
182  cv::Mat imageConv;
183  auto dataType = CV_MAKETYPE(cv::DataType<T>::depth, image.channels());
184  if(image.type() == dataType) {
185  imageConv = image;
186  } else {
187  image.assignTo(imageConv, dataType);
188  }
189 
190  slicer_ = PyramidImageSlicer(imageConv, pyramidSlidingWindow, windowSize);
191 }

209 template <typename T>
210 std::vector<cv::Mat> ImageBatch<T>::imageChannels(size_t index) const
211 {
212  if(index >= windows.size()) {
213  DG_ERROR_THROW("Image index out of bounds");
214  }
215 
216  if(!data || !dataSize) {
217  DG_ERROR_THROW("Image data is null");
218  }
219 
220  const auto& window = windows[index].window;
221  auto dataType = CV_MAKETYPE(cv::DataType<T>::depth, 1);
222 
223  std::vector<cv::Mat> channels;
224  for(int channelIdx = 0; channelIdx < numChannels; ++channelIdx) {
225  void* channelData = data + index * window.size().area() * numChannels + window.size().area() * channelIdx;
226  cv::Mat channel(window.size(), dataType, channelData);
227 
228  channels.emplace_back(std::move(channel));
229  }
230 
231  return std::move(channels);
232 }

376 template <typename T>
377 void PyramidImageBatchIterator<T>::calcBatch()
378 {
379  if(!batcher_ || slicer_ == batcher_->slicer().end()) {
380  batch_ = ImageBatch<T>();
381  return;
382  }
383 
384  batch_.windows.clear();
385 
386  size_t i = 0;
387  while(slicer_ != batcher_->slicer().end() && i++ < batcher_->batchSize()) {
388  const auto& pyramidImage = *slicer_;
389 
390  // Convert the window dimensions from being relative to the pyramid level window size to being
391  // relative to the original image size
392  auto window = pyramidImage.window;
393  const auto& curSize = pyramidImage.level.size;
394  const auto& topSize = batcher_->slicer().image().size();
395  window.x = (int)ceilf((float)topSize.width / curSize.width * window.x);
396  window.y = (int)ceilf((float)topSize.height / curSize.height * window.y);
397  window.width = (int)ceilf((float)topSize.width / curSize.width * window.width);
398  window.height = (int)ceilf((float)topSize.height / curSize.height * window.height);
399 
400  batch_.windows.emplace_back(PyramidWindow(pyramidImage.level, window));
401 
402  const auto& image = pyramidImage.image;
403  cv::Mat outImage;
404 
405  if(batcher_->preprocessFunc()) {
406  batcher_->preprocessFunc()(outImage, image);
407  } else {
408  outImage = image;
409  }
410 
411  // OpenCV stores images as BIP (Band Interleaved by Pixel), Caffe wants them as BSQ (Band Sequential),
412  // so we need to split image channels, writing the output directly into our batch data.
413  auto channels = batch_.imageChannels(i - 1);
414  cv::split(outImage, channels);
415 
416  ++slicer_;
417  }
418 }
