50 std::vector<option> ignore_numbers(std::vector<std::string>& args)
51 {
52  std:: vector<option> result;
53  int pos = 0;
54  while(!args.empty()) {
55  const auto& arg = args[0];
56  double num;
57  if(conversion::try_lexical_convert(arg, num)) {
58  result.push_back(option());
59  option& opt = result.back();
60 
61  opt.position_key = pos++;
62  opt.value.push_back(arg);
63  opt.original_tokens.push_back(arg);
64 
65  args.erase(args.begin());
66  } else {
67  break;
68  }
69  }
70 
71  return result;
72 }
