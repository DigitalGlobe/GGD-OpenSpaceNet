64 typename std::enable_if<!std::numeric_limits<T>::is_integer, bool>::type
65 almostEq(T x, T y, int ulp=4)
66 {
67  // the machine epsilon has to be scaled to the magnitude of the values used
68  // and multiplied by the desired precision in ULPs (units in the last place)
69  return std::abs(x-y) < std::numeric_limits<T>::epsilon() * std::abs(x+y) * ulp
70  // unless the result is subnormal
71  || std::abs(x-y) < std::numeric_limits<T>::min();
72 }
